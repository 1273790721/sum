ajax加载树状列表说明：

1.在页面引入easyui插件及css。

2.页面加载时，初始化ajax树。 

$(function(){
    initTable();
    $(window).resize(function(){
        initTable();
    });
})
function initTable() {
    $('#tree').treegrid({
        url: "{:url('tree/ajax_tree')}",//这里为后台请求地址
        rownumbers: true,
        idField: 'i',
        animate:true,
        treeField: 'tree_name',
        columns: [
            [{
                field: 'tree_name',
                title: '书名',
                align: 'left',
                width: "100%"
            }]
        ]
    });
}

3.后台处理数据，以json格式返回给前台。例如： 

$return = 
[
    ['i'=>1,'tree_name'=>'西游记',
    'children'=>[['i'=>2,'tree_name'=>'第一章'],['i'=>3,'tree_name'=>'第二章']]],
    ['i'=>4,'tree_name'=>'三国演义',
    'children'=>[['i'=>5,'tree_name'=>'第一章'],['i'=>6,'tree_name'=>'第二章']]]
];
echo json_encode($return);
